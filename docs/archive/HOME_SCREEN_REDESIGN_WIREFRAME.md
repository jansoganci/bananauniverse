```sql
-- Only show seasonal category during relevant months
SELECT * FROM categories
WHERE id = 'seasonal'
  AND (
    (EXTRACT(MONTH FROM now()) IN (10, 11) AND name LIKE '%Thanksgiving%') OR
    (EXTRACT(MONTH FROM now()) IN (11, 12, 1) AND name LIKE '%Christmas%')
  );
```
